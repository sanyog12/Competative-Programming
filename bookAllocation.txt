#include <bits/stdc++.h> 

bool possible(vector<int> arr, int n, int m, int mid){

    int StudentNumber = 1;

    int pageCount = 0;

    for(int i = 0; i < n; i++){

        if(pageCount + arr[i] <= mid){

            pageCount += arr[i];

        }

        else{

            StudentNumber++;

            if(StudentNumber > m || arr[i] > mid){

                return 0;

            }

            pageCount = arr[i];

        }

    }

    return 1;

}

int allocateBooks(vector<int> arr, int n, int m) {

    // Write your code here.
    int s = 0; 

    int e = 0;

    for(int i = 0; i < n; i++){

        e += arr[i];

    }

    int ans = -1;

    int mid = s+(e-s)/2;

    while(s <= e){

        if(possible(arr,n,m,mid)){

            ans = mid;

            e = mid - 1;

        }

        else{

            s = mid + 1;

        }

        mid = s+(e-s)/2;

    }

    return ans;

}
